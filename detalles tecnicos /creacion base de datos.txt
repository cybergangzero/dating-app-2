 CREATE TABLE users(
 username varchar(20),
 password varchar(60),
 name varchar(20),
 lastName varchar(20),
 sex bool,
 date_of_birth date,
 age int,
 country varchar(20),
 premium bool,
 maritalStatus varchar(20),
 sons bool,
 housingSituation varchar(15),
 educationalLevel varchar(20),
 work bool,
 ethnicGroup varchar(20),
 smokes bool,
 drink bool,
 heigth float,
 bodyType varchar(15),
 header varchar(50),
 description varchar(300),
 online bool,
 src_profile_photo varchar(100),
 recovery_code varchar(10)
 );

create table conversations(
 id serial,
 id_user_a varchar(15),
 id_user_b varchar(15),
 last_message_sent text
);

create table messages(
 id_conversation int,
 id_user varchar(15),
 message text
);

create table new_messages(
 id_user varchar(15),
 id_conversation int,
 amount int
);

create table likes(
 id_user varchar(15),
 id_user_who_likes varchar(15),
 checked_by_the_user bool
);
